// server.js const express = require("express"); const path = require("path"); const cors = require("cors"); const { Pool } = require("pg"); const app = express(); const PORT = process.env.PORT || 3000; const db = new Pool({ connectionString: process.env.DATABASE_URL, ssl: { rejectUnauthorized: false } }); app.use(cors()); app.use(express.json()); app.use(express.static(path.join(__dirname, "public"))); // API routes app.use("/api/profile", require("./api/profile")(db)); app.use("/api/group", require("./api/group")(db)); app.use("/api/sos", require("./api/sos")(db)); app.use("/api/preferences", require("./api/preferences")(db)); // SPA fallback app.get("*", (req, res) => { res.sendFile(path.join(__dirname, "public", "index.html")); }); app.listen(PORT, () => { console.log(`TogetherSafe backend running on port ${PORT}`); });
